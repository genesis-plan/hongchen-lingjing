### 2. 代码实现
```javascript
// 多API自动切换
const apis = {
  'aliyun': { key: process.env.ALI_KEY, base: 'dashscope.aliyuncs.com' },
  'baidu': { key: process.env.BAIDU_KEY, base: 'aip.baidubce.com' },
  'deepseek': { key: process.env.DEEPSEEK_KEY, base: 'api.deepseek.com' }
};

async function callLLM(prompt) {
  for (const [name, config] of Object.entries(apis)) {
    try {
      return await callAPI(config, prompt);
    } catch (e) {
      console.log(`${name}失败，切换到下一个`);
      continue;
    }
  }
  throw new Error('所有API都失败');
}
```

### 3. 成本控制
- 优先使用免费额度
- 启用缓存，避免重复请求
- 使用小模型处理简单任务
- 大模型仅用于复杂任务
